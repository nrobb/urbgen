 46 /**
 47  * Defines a polygon.
 48  * @constructor
 49  * @param {URBGEN.Point} p0 - the top left corner of the polygon.
 50  * @param {URBGEN.Point} p1 - the top right corner of the polygon.
 51  * @param {URBGEN.Point} p2 - the bottom left corner of the polygon.
 52  * @param {URBGEN.Point} p3 - the bottom right corner of the polygon.
 53  */
 54 URBGEN.Poly = function (p0, p1, p2, p3) {
 55     /**
 56      * This polygon's corners, in the order: topleft, topright, bottomleft,
 57      * bottomright.
 58      * @type URBGEN.Point[]
 59      */
 60     this.corners = [p0, p1, p2, p3];
 61     /**
 62      * The lengths of this polygon's edges, in the order: top, right, left,
 63      * bottom.
 64      * @type number[]
 65      */
 66     this.edgeLengths = [
 67         URBGEN.Util.getLineSegmentLength(this.corners[0], this.corners[
 68             1]),
 69         URBGEN.Util.getLineSegmentLength(this.corners[1], this.corners[
 70             3]),
 71         URBGEN.Util.getLineSegmentLength(this.corners[0], this.corners[
 72             2]),
 73         URBGEN.Util.getLineSegmentLength(this.corners[2], this.corners[
 74             3])
 75     ];
 76     /**
 77      * The vertical angle of this polygon's grid.
 78      * @type number
 79      */
 80     this.gridAngle = undefined;
 81 };
 82 /**
 83  * Sets this polygon's grid angle.
 84  * @param {number} random - A random number between 0 - 1 (inclusive).
 85  */
 86 URBGEN.Poly.prototype.setGridAngle = function (random) {
 87     var start = URBGEN.Util.linearInterpolate(this.corners[0], this.corners[
 88             2],
 89         0.5);
 90     var end = URBGEN.Util.linearInterpolate(this.corners[1], this.corners[3],
 91         random * 0.5 + 0.2);
 92     this.gridAngle = URBGEN.Util.getAngle(start, end);
 93 };
 94 /**
 95  * Sets this polygon's corners as neighbors so that there are no
 96  * extra points included on any edge.
 97  */
 98 URBGEN.Poly.prototype.makeSimple = function () {
 99     this.corners[0].neighbors[2] = this.corners[2];
100     this.corners[0].neighbors[3] = this.corners[1];
101     this.corners[1].neighbors[1] = this.corners[0];
102     this.corners[1].neighbors[2] = this.corners[3];
103     this.corners[2].neighbors[0] = this.corners[0];
104     this.corners[2].neighbors[3] = this.corners[3];
105     this.corners[3].neighbors[0] = this.corners[1];
106     this.corners[3].neighbors[1] = this.corners[2];
107 };

325 /**
326  * Sets this generator's parameters. If any paramaters in the cityParmas object
327  * are undefined, sets those parameters to random values.
328  * @param {Object} cityParams - The paramters this generator should use to
329  *     generate a city.
330  */
331 URBGEN.Generator.prototype.setParams = function (cityParams) {
332     if(cityParams === undefined) {
333         cityParams = {};
334     }
335     if((0 > cityParams.localGrids) || (1 < cityParams.localGrids) || (
336         cityParams.localGrids === undefined)) {
337         this.localGrids = Math.random();
338     } else {
339         this.localGrids = cityParams.localGrids;
340     }
341     if((0 > cityParams.randomSeed) || (1 < cityParams.randomSeed) || (
342         cityParams.randomSeed === undefined)) {
343         this.randomSeed = Math.random();
344     } else {
345         this.randomSeed = cityParams.randomSeed;
346     }
347     if((0 > cityParams.globalAngle) || (1 < cityParams.globalAngle) || (
348         cityParams.globalAngle === undefined)) {
349         this.globalAngle = Math.random();
350     } else {
351         this.globalAngle = cityParams.globalAngle;
352     }
353     if((0 > cityParams.throughRoads) || (URBGEN.Constants.MAX_THROUGH_ROADS <
354         cityParams.throughRoads) || (cityParams.throughRoads ===
355         undefined)) {
356         this.throughRoads = Math.random() * URBGEN.Constants.MAX_THROUGH_ROADS;
357     } else {
358         this.throughRoads = cityParams.throughRoads;
359     }
360     this.blockSize = cityParams.blockSize || Math.random() * (URBGEN.Constants
361             .MAX_BLOCK_SIZE - URBGEN.Constants.MIN_BLOCK_SIZE) + URBGEN.Constants
362         .MIN_BLOCK_SIZE;
363     this.cityWidth = cityParams.cityWidth || Math.random() * (URBGEN.Constants
364             .MAX_CITY_WIDTH - URBGEN.Constants.MIN_CITY_WIDTH) + URBGEN.Constants
365         .MIN_CITY_WIDTH;
366     this.cityDepth = cityParams.cityDepth || Math.random() * (URBGEN.Constants
367             .MAX_CITY_DEPTH - URBGEN.Constants.MIN_CITY_DEPTH) + URBGEN.Constants
368         .MIN_CITY_DEPTH;
369     this.streetWidth = cityParams.streetWidth || Math.random() * (URBGEN.Constants
370             .MAX_STREET_WIDTH - URBGEN.Constants.MIN_STREET_WIDTH) + URBGEN
371         .Constants.MIN_STREET_WIDTH;
372 };

390 /**
391  * Initializes this generator.
392  * @throws {URBGEN.Exception.InvalidParamException}
393  */
394 URBGEN.Generator.prototype.init = function () {
395     if(isNaN(this.localGrids) || isNaN(this.randomSeed) || isNaN(this.globalAngle) ||
396         isNaN(this.blockSize) || isNaN(this.cityWidth) || isNaN(this.cityDepth) ||
397         isNaN(this.throughRoads) || isNaN(this.streetWidth)) {
398         throw new URBGEN.Exception.InvalidParamException();
399     }
400     // Initialise the polygon array
401     this.cityPolys = [];
402     // Seed the random number generator
403     this.random.seed = this.randomSeed;
404     // Instantiate a new city
405     this.city = new URBGEN.City();
406     // Build the initial polygon
407     var topLeft = new URBGEN.Point(0, 0, 0);
408     var topRight = new URBGEN.Point(this.cityWidth, 0, 0);
409     var bottomLeft = new URBGEN.Point(1, this.cityDepth, 0);
410     var bottomRight = new URBGEN.Point(this.cityWidth - 1, this.cityDepth,
411         0);
412     var poly = new URBGEN.Poly(topLeft, topRight, bottomLeft, bottomRight);
413     poly.makeSimple();
414     this.cityPolys.push(poly);
415     // Add the corners to the nodes graph
416     this.nodes = [topLeft, topRight, bottomLeft, bottomRight];
417     // Set the city's polygon
418     this.city.poly = poly;
419     // Set the city's area
420     this.city.area = URBGEN.Util.areaPoly(poly);
421     // Set the regularity params
422     this.regularity1 = this.globalAngle * (URBGEN.Constants.MAX_REGULARITY -
423         URBGEN.Constants.MIN_REGULARITY) + URBGEN.Constants.MIN_REGULARITY;
424     this.regularity2 = URBGEN.Constants.MAX_REGULARITY - this.globalAngle *
425         (URBGEN.Constants.MAX_REGULARITY - URBGEN.Constants.MIN_REGULARITY);
426 };
427 /**
428  * Recursively processes an array of polygons.
429  * @param {URBGEN.Point[]} polys - The array of polygons to be processed.
430  * @return {URBGEN.Poly[]} The new polygons.
431  */
432 URBGEN.Generator.prototype.processPolyRecursively = function (polys) {
433     var newPolys = [];
434     for(var i = 0; i < polys.length; i++) {
435         newPolys = newPolys.concat(this.processPoly(polys[i]));
436     }
437     if(polys.length !== newPolys.length) {
438         return this.processPolyRecursively(newPolys);
439     }
440     return newPolys;
441 };
442 /**
443  * Generates a city.
444  * @param {Object} cityParams - The parameters that should be used to generate
445  *     the city.
446  * @return {URBGEN.City} The generated city.
447  */
448 URBGEN.Generator.prototype.generate = function (cityParams) {
449     this.setParams(cityParams);
450     this.init();
451     // Get the initial polygons
452     this.cityPolys = this.processPolyRecursively(this.cityPolys);
453     // Build the city blocks.
454     for(var i = 0; i < this.cityPolys.length; i++) {
455         //TODO this doesn't add every node yet
456         this.city.roads.push(this.cityPolys[i].corners[0]);
457         var poly = this.cityPolys[i];
458         var block = new URBGEN.Block(poly, this.streetWidth);
459         this.city.blocks.push(block);
460     }
461     // Build the plots
462     for(var j = 0; j < this.city.blocks.length; j++) {
463         var plots = [];
464         this.plotBuilder.poly = this.city.blocks[j].innerPoly;
465         var plotPolys = this.director.execute(this.plotBuilder);
466         for(var k = 0; k < plotPolys.length; k++) {
467             var plot = new URBGEN.Plot(plotPolys[k]);
468             plot.height = this.random.next() * 20 + 20;
469             plots.push(plot);
470         }
471         this.city.blocks[j].plots = plots;
472     }
473     // Build the 3D geometry
474     this.buildGeometry();
475     //return the city
476     return this.city;
477 };
478 /**
479  * Builds a 3D geometry for this generator's current city.
480  */
481 URBGEN.Generator.prototype.buildGeometry = function () {
482     var vertices = this.city.geometry.vertices;
483     var faces = this.city.geometry.faces;
484     var plots = this.city.getPlots();
485     for(var i = 0; i < plots.length; i++) {
486         var building = plots[i].poly;
487         var height = plots[i].height;
488         for(var j = 0; j < building.corners.length; j++) {
489             var vertexTop = building.corners[j];
490             vertices.push([vertexTop.x, vertexTop.y, height]);
491         }
492         for(var k = 0; k < building.corners.length; k++) {
493             var vertexBottom = building.corners[k];
494             vertices.push([vertexBottom.x, vertexBottom.y, 0]);
495         }
496     }
497     for(var l = 1; l < vertices.length + 1; l += 8) {
498         faces.push([l, l + 1, l + 2]);
499         faces.push([l + 2, l + 1, l + 3]);
500         faces.push([l, l + 1, l + 4]);
501         faces.push([l + 4, l + 1, l + 5]);
502         faces.push([l + 1, l + 3, l + 5]);
503         faces.push([l + 5, l + 3, l + 7]);
504         faces.push([l + 3, l + 2, l + 7]);
505         faces.push([l + 7, l + 2, l + 6]);
506         faces.push([l + 2, l, l + 6]);
507         faces.push([l + 6, l, l + 4]);
508     }
509 };
510 /**
511  * Processes a polygon. If a polygon cannot be processed, it is returned.
512  * @param {URBGEN.Poly} poly - The polygon to process.
513  * @return {URBGEN.Poly[]} The new polygons, or the original poly.
514  */
515 URBGEN.Generator.prototype.processPoly = function (poly) {
516     if(URBGEN.Util.areaPoly(poly) < this.blockSize) {
517         return [poly];
518     }
519     this.prepare(poly);
520     try {
521         var newPolys = this.director.execute(this.builder, this.center);
522         return newPolys;
523     } catch(error) {
524         if(error instanceof URBGEN.Exception.EdgeTooShortException) {
525             return [poly];
526         }
527     }
528 };
529 /**
530  * Prepares to process a polygon by loading the appropriate Builder.
531  * @param {URBGEN.Poly} poly - The polygon being prepared for.
532  */
533 URBGEN.Generator.prototype.prepare = function (poly) {
534     // Set the correct builder
535     var horizontalSides = poly.edgeLengths[0] + poly.edgeLengths[3];
536     var verticalSides = poly.edgeLengths[1] + poly.edgeLengths[2];
537     if(verticalSides > horizontalSides) {
538         this.builder = this.horizontalBuilder;
539     } else {
540         this.builder = this.verticalBuilder;
541     }
542     this.builder.poly = poly;
543 };
544 /**
545  * Returns a THREE.Shape representing the specified poly.
546  * @param {URBGEN.Poly} poly - The poly providing the coordinates.
547  * @param {THREE.Shape} shape - The shape to have its coordinates set.
548  * @return {THREE.Shape} The shape with coordinates set.
549  */
550 URBGEN.Generator.prototype.buildThreeShape = function (poly, shape) {
551     shape.moveTo(poly.corners[0].x, poly.corners[0].y);
552     shape.lineTo(poly.corners[1].x, poly.corners[1].y);
553     shape.lineTo(poly.corners[3].x, poly.corners[3].y);
554     shape.lineTo(poly.corners[2].x, poly.corners[2].y);
555     shape.lineTo(poly.corners[0].x, poly.corners[0].y);
556     return shape;
557 };

576     }

596 /**
597  * Represents a builder.
598  * @constructor
599  * @param {URBGEN.Generator} generator - The generator that owns this builder.
600  */
601 URBGEN.Builder = function (generator) {
602     /**
603      * The generator that owns this builder.
604      * @type URBGEN.Generator
605      */
606     this.generator = generator;
607     /**
608      * The current polygon this builder is operating on.
609      * @type URBGEN.Poly
610      */
611     this.poly = undefined;
612     /**
613      * Ths origin point for this builder's dividing line.
614      * @type URBGEN.Point
615      */
616     this.origin = undefined;
617     /**
618      * Ths end point for this builder's dividing line.
619      * @type URBGEN.Point
620      */
621     this.endPoint = undefined;
622     /**
623      * The points this builder will use to build new polygons.
624      * @type URBGEN.Point[]
625      */
626     this.newPoints = [];
627 };
628 /**
629  * Returns an array of new polygons created from this builder's current points.
630  * @return {URBGEN.Poly[]} The new polygons.
631  * @throws {URBGEN.Exception.EdgeTooShortException}
632  */
633 URBGEN.Builder.prototype.buildPolys = function () {
634     var polys = [];
635     for(var i = 0; i < this.newPoints.length; i++) {
636         // Build the new polygon
637         var points = this.newPoints[i];
638         var poly = new URBGEN.Poly(points[0], points[1], points[2], points[
639             3]);
640         // Set the grid angle of the new polygon, if needed
641         var localGridThreshold = this.generator.localGrids * this.generator
642             .city.area;
643         if(URBGEN.Util.areaPoly(poly) < localGridThreshold) {
644             if(this.poly.gridAngle === undefined) {
645                 poly.setGridAngle(this.generator.random.next());
646             } else {
647                 poly.gridAngle = this.poly.gridAngle;
648             }
649         }
650         // Add the polygon
651         polys.push(poly);
652     }
653     for(var j = 0; j < polys.length; j++) {
654         for(var k = 0; k < polys[j].edgeLengths.length; k++) {
655             var length = polys[j].edgeLengths[k];
656             if(length < this.minEdgeLength) {
657                 throw new URBGEN.Exception.EdgeTooShortException();
658             }
659         }
660     }
661     return polys;

663 /**
664  * Sets this builder's origin and end points for the new dividing line.
665  * @throws {URBGEN.Exception.EdgeTooShortException}

667 URBGEN.Builder.prototype.setUp = function () {
668     var edgeStart = this.poly.corners[0];
669     var edgeEnd = this.poly.corners[this.corners[0]];

672     var endPoint;

693 /**
694  * Returns the point at which a line at the specified angle intersects with the
695  * edge defined by the specified start and end points. If this point does not lie
696  * on the legal part of the edge (defined as a region starting at the minimum
697  * edge length from the start point and ending at the minium edge length before
698  * the end point) then the closest point in the legal region is returned.
699  * @param {URBGEN.Point} edgeStart - The start point of the edge.
700  * @param {URBGEN.Point} edgeEnd - The end point of the edge.
701  * @param {number} angle - The angle in radians of the intersecting line.
702  * @return {URBGEN.Point} The closest legal point at which the intersecting line
703  *     meets the edge.
704  * @throws {URBGEN.Exception.EdgeTooShortException}
705  */
706 URBGEN.Builder.prototype.pointByAngle = function (edgeStart, edgeEnd, angle) {
707     // Get the edge's length
708     var length = URBGEN.Util.getLineSegmentLength(edgeStart, edgeEnd);
709     // Throw an error if the edge is too short
710     if(length <= URBGEN.Constants.MIN_EDGE_LENGTH) {
711         throw new URBGEN.Exception.EdgeTooShortException();
712     }
713     var minR = URBGEN.Constants.MIN_EDGE_LENGTH / length;
714     var edgeAngle = URBGEN.Util.getAngle(edgeStart, edgeEnd);
715     var point = URBGEN.Util.getIntersect(edgeStart, edgeAngle, this.origin,
716         angle);
717     var r = URBGEN.Util.getPointAsRatio(point, edgeStart, edgeEnd);
718     if(r < minR)
719         r = minR;
720     if(r > 1 - minR)
721         r = 1 - minR;
722     return this.pointByRValue(edgeStart, edgeEnd, r);
723 };
724 /**
725  * Returns the point represented by the specified r value on the edge defined by
726  * the specified start and end points. If this point does not lie on the legal
727  * part of the edge (defined as a region starting at the minimum edge length from
728  * the start point and ending at the minium edge length before the end point) then
729  * the closest point in the legal region is returned.
730  * @param {URBGEN.Point} edgeStart - The start point of the edge.
731  * @param {URBGEN.Point} edgeEnd - The end point of the edge.
732  * @param {number} r - The r value of the new point.
733  * @return {URBGEN.Point} The closest legal point at which the intersecting line
734  *     meets the edge.
735  * @throws {URBGEN.Exception.EdgeTooShortException}
736  */
737 URBGEN.Builder.prototype.pointByRValue = function (edgeStart, edgeEnd, r) {
738     // Get the edge's length
739     var length = URBGEN.Util.getLineSegmentLength(edgeStart, edgeEnd);
740     // Throw an error if the edge is too short
741     if(length <= URBGEN.Constants.MIN_EDGE_LENGTH) {
742         throw new URBGEN.Exception.EdgeTooShortException();
743     }
744     // Work out the legal part of the edge as a range (0 < range < 1)
745     var minR = URBGEN.Constants.MIN_EDGE_LENGTH / length;
746     var range = 1 - 2 * minR;
747     // Use the regularity1 to find a point on this range
748     var pointR = range * r + minR;
749     // Get the actual point
750     var point = URBGEN.Util.linearInterpolate(edgeStart, edgeEnd, pointR);
751     return point;
752 };
753 /**
754  * Adds the specified point to the edge with the specified start and end points.
755  * If there is already a point on this edge within the current throughRoad
756  * distance, returns that point. Otherwise, finds the new point's neighbors on
757  * the edge, sets the neighbor relations to include the new point, and returns
758  * that point.
759  * @param {URBGEN.Point} point - The point to be added to the edge.
760  * @param {URBGEN.Point} edgeStart - The start point of the edge.
761  * @param {URBGEN.Point} edgeEnd - The end point of the edge.
762  * @return {URBGEN.Point} A point within the throughRoad distance on the edge,
763  *     or the originally specified point (with neighbor relations set).
764  */
765 URBGEN.Builder.prototype.addPointToPath = function (point, edgeStart, edgeEnd) {
766     var path = URBGEN.Util.getDirectedPath(edgeStart, edgeEnd, this.direction);
767     var distance = this.generator.throughRoads;
768     var nearPoint = URBGEN.Util.checkNearPoints(point, path, distance,
769         false);
770     if(nearPoint === point) {
771         var neighbors = URBGEN.Util.getNeighbors(point, path);
772         URBGEN.Util.insertPoint(point, neighbors.prev, neighbors.nxt);
773     }
774     return nearPoint;
775 };

1029 /**
1030  * Returns a point on the line segment p0p1 which is the specified length along
1031  * the line.
1032  * @param {URBGEN.Point} p0 - The start point of the line.
1033  * @param {URBGEN.Point} p1 - The end point of the line.
1034  * @param {number} length - The length along the line of the new point.
1035  * @return {URBGEN.Point} The new point.

1038     var totalLength = URBGEN.Util.getLineSegmentLength(p0, p1);
1039     if(length > totalLength) {
1040         return p1;
1041     }

1045 /**
1046  * Finds a point on the line segment p0p1 using the r value (ratio of the line).
1047  * @param {URBGEN.Point} p0 - The start point of the line.
1048  * @param {URBGEN.Point} p1 - The end point of the line.

1050  * @return {URBGEN.Point} The new point.
1051  */

1053     var x = (1 - r) * p0.x + r * p1.x;
1054     var y = (1 - r) * p0.y + r * p1.y;
1055     var z = (1 - r) * p0.z + r * p1.z;
1056     return new URBGEN.Point(x, y, z);

1060  * @param {URBGEN.Point} p0 - The start point of the line segment.

1072         } else {

1099 /**

1101  * through p0 and p1, relative to the line segment p0p1.
1102  * @param {URBGEN.Point} point - The point.
1103  * @param {URBGEN.Point} p0 - A point on the line.

1109     var d2;
1110     // If the line is parallel with the y-axis, use the difference in y values

1112         d1 = point.y - p0.y;
1113         d2 = p1.y - p0.y;

1115         d1 = point.x - p0.x;
1116         d2 = p1.x - p0.x;

1120 /**

1122  * @param {URBGEN.Poly} poly - The polygon.
1123  * @return {number} The polygon's area.

1126     var x0 = poly.corners[3].x - poly.corners[0].x;

1133 /**

1135  * direction. If p1 is not found in maxSteps iterations, returns false. If
1136  * maxSteps is not specified, defaults to 1000.
1137  * @param {URBGEN.Point} p0 - The start point of the path.
1138  * @param {URBGEN.Point} p1 - The end point of the path.
1139  * @param {number} direction - The direction of the path.
1140  * @param {number} maxSteps - The maximum number of iterations.
1141  * @return {URBGEN.Point[]} The path from p0 t0 p1.
1142  */
1143 URBGEN.Util.getDirectedPath = function (p0, p1, direction, maxSteps) {
1144     if (maxSteps === undefined) {
1145         maxSteps = 1000;
1146     }
1147     var points = [p0];
1148     var i = 0;
1149     while(points[i] !== p1) {
1150         points.push(points[i].neighbors[direction]);
1151         i++;
1152         if(i === maxSteps)
1153             return undefined;
1154     }
1155     return points;

1157 /**
1158  * Given two lines, defined by a point on the line and the angle of the line,
1159  * returns the point at which the two lines intersect. If the lines are colinear,
1160  * returns p1.
1161  * @param {URBGEN.Point} p0 - The start point of the first line.
1162  * @param {number} a0 - The angle in radians of the first line.
1163  * @param {URBGEN.Point} p1 - The start point of the second line.
1164  * @param {number} a1 - The angle in radians of the second line.
1165  * @return {URBGEN.Point} The intersection of the two lines.
1166  */
1167 URBGEN.Util.getIntersect = function (p0, a0, p1, a1) {
1168     var m0 = Math.tan(a0);
1169     var m1 = Math.tan(a1);
1170     var x;
1171     var y;
1172     var point;
1173     // Check if the lines are colinear
1174     if(m0 === m1)
1175         return p0;
1176     // Check if either line is colinear with the y axis
1177     if(m0 === Infinity) {
1178         x = p0.x;
1179         y = x * m1 + (p1.y - m1 * p1.x);
1180     } else if(m1 === Infinity) {
1181         x = p1.x;
1182         y = x * m0 + (p0.y - m0 * p0.x);
1183         // Otherwise, find the intersection point
1184     } else {
1185         x = (p1.y - p0.y + (m0 * p0.x) - (m1 * p1.x)) / (m0 - m1);
1186         y = m1 * (x - p1.x) + p1.y;
1187     }
1188     point = new URBGEN.Point(x, y, 0);
1189     return point;

1191 /**
1192  * Returns a path of equidistant points on the linesegment p0p1. Note that the
1193  * distance between the second last point and the end point (p1) can be of
1194  * any length.
1195  * @param {URBGEN.Point} p0 - The start point of the line segment.
1196  * @param {URBGEN.Point} p1 - The end point of the line segment.
1197  * @param {number} length - The distance between each point in the path.
1198  * @return {URBGEN.Point[]} The path of equidistant points.
1199  * @throws {URBGEN.Exception.PointsNotNeighborsException}
1200  */
1201 URBGEN.Util.divideLine = function (p0, p1, length) {
1202     if(p0.neighbors.indexOf(p1) === -1) {
1203         throw new URBGEN.Exception.PointsNotNeighborsException();
1204     }
1205     var points = [p0];
1206     var i = 0;
1207     var currentLineLength = URBGEN.Util.getLineSegmentLength(points[i], p1);
1208     while(length * 2 <= currentLineLength) {
1209         points.push(URBGEN.Util
1210             .linearInterpolateByLength(points[i], p1, length));
1211         i++;
1212         currentLineLength = URBGEN.Util.getLineSegmentLength(points[i], p1);
1213     }
1214     points.push(p1);
1215     return points;
1216 };

1271 /**
1272  * Returns an inward offset of the specified polygon. That is, the returned
1273  * polygon is formed by insetting the edges of the specified polygon.
1274  * @param {URGBEN.Poly} poly - The poly to offset.
1275  * @param {number} length - The distance of the inward offset.
1276  * @return {URBGEN.Poly} The offset polygon.
1277  */
1278 URBGEN.Util.insetPoly = function (poly, length) {
1279     // Get the edges
1280     var edges = [
1281         [poly.corners[0], poly.corners[1]],
1282         [poly.corners[1], poly.corners[3]],
1283         [poly.corners[3], poly.corners[2]],
1284         [poly.corners[2], poly.corners[0]]
1285     ];
1286     // Get the offset edges and angles
1287     var offsetEdges = [];
1288     var angles = [];
1289     for(var i = 0; i < edges.length; i++) {
1290         offsetEdges.push(URBGEN.Util.offsetLineSegment(edges[i][0],
1291             edges[i][1], length));
1292         angles.push(URBGEN.Util.getAngle(offsetEdges[i][0], offsetEdges[i][
1293             1
1294         ]));
1295     }
1296     // Find the new corners
1297     var tl = URBGEN.Util.getIntersect(offsetEdges[0][0], angles[0],
1298         offsetEdges[3][0], angles[3]);
1299     var tr = URBGEN.Util.getIntersect(offsetEdges[1][0], angles[1],
1300         offsetEdges[0][0], angles[0]);
1301     var bl = URBGEN.Util.getIntersect(offsetEdges[3][0], angles[3],
1302         offsetEdges[2][0], angles[2]);
1303     var br = URBGEN.Util.getIntersect(offsetEdges[2][0], angles[2],
1304         offsetEdges[1][0], angles[1]);

1307 /**
1308  * Sets the neighbor relations of p0 and p1 with the newPoint. If p0 and p1 are
1309  * not neighbors, returns false, otherwise true.
1310  * @param {URBGEN.Point} newPoint - The new point.
1311  * @param {URBGEN.Point} p0 - The first neighbor point.
1312  * @param {URNGEN.Point} p1 - The second neighbor point.
1313  * @return {boolean} True if successful, false otherwise.
1314  */
1315 URBGEN.Util.insertPoint = function (newPoint, p0, p1) {
1316     var direction = p0.neighbors.indexOf(p1);
1317     var oppDirection = (direction + 2) % 4;
1318     p1.neighbors[oppDirection] = newPoint;
1319     p0.neighbors[direction] = newPoint;
1320     newPoint.neighbors[direction] = p1;
1321     newPoint.neighbors[oppDirection] = p0;
1322     return true;
1323 };
1324 /**
1325  * Returns the specified the two points closest in opposite directions to the
1326  * specified point.
1327  * @param {URBGEN.Point} point - The point to find neighbors for.
1328  * @param {URBGEN.Point[]} - The path of potential neighbors.
1329  * @return {Object} The prev and nxt points to the specified point.
1330  * @throws {URBGEN.Exception.PointOutOfRangeException}
1331  */
1332 URBGEN.Util.getNeighbors = function (point, points) {
1333     var neighbors = {
1334         prev: undefined,
1335         nxt: undefined
1336     };
1337     if(points.length === 2) {
1338         neighbors.prev = points[0];
1339         neighbors.nxt = points[1];
1340         return neighbors;
1341     }
1342     // Find the point as a ratio of the line
1343     var pointR = URBGEN.Util.getPointAsRatio(point, points[0],
1344         points[points.length - 1]);
1345     // if the point lies beyond either end of the path, throw error
1346     if(pointR <= 0 || pointR >= 1) {
1347         throw new URBGEN.Exception.PointOutOfRangeException();
1348     }
1349     for(var i = 1; i < points.length; i++) {
1350         var currPoint = points[i];
1351         var r = URBGEN.Util.getPointAsRatio(currPoint, points[0],
1352             points[points.length - 1]);
1353         //TODO what if r === pointR? ie, the point is identical to one on the path?
1354         if(r > pointR) {
1355             neighbors.prev = points[i - 1];
1356             neighbors.nxt = points[i];
1357             return neighbors;
1358         }
1359     }
1360     return false;
1361 };
1362 /**
1363  * Returns a point on the specified edge that is within the specified distance
1364  * of the specified point. If includeEnds is true, then the start and end points
1365  * of the edge will be included in the search. If no such point exists, returns
1366  * the original point.
1367  * @param {URBGEN.Point} point - The point to find near points for.
1368  * @param {URBGEN.Point[]} - The path of potential near points.
1369  * @param {number} distance - The distance within which near points can be found.
1370  * @param {boolean} includeEnds - true if the start and end points should be
1371  *     included, false if not.
1372  * @return {URBGEN.Point} The closest point to the specified point, or the
1373  *     original point is no such near point exists.
1374  */
1375 URBGEN.Util.checkNearPoints = function (point, points, distance, includeEnds) {
1376     var neighbors = URBGEN.Util.getNeighbors(point, points);
1377     var d0 = Math.abs(URBGEN.Util.getLineSegmentLength(neighbors.prev,
1378         point));
1379     var d1 = Math.abs(URBGEN.Util.getLineSegmentLength(point, neighbors.nxt));
1380     if(d0 < distance && d0 <= d1) {
1381         if(neighbors.prev === points[0]) {
1382             if(includeEnds) {
1383                 return points[0];
1384             }
1385         } else {
1386             return neighbors.prev;
1387         }
1388     } else if(d1 < distance) {
1389         if(neighbors.nxt === points[points.length - 1]) {
1390             if(includeEnds) {
1391                 return points[points.length - 1];
1392             }
1393         } else {
1394             return neighbors.nxt;
1395         }
1396     }
1397     return point;
1398 };
1399 /**
1400  * Returns the point in points that has the shortest straight line distance to
1401  * target. If any points have equal distances to target, returns the point which
1402  * comse first in points.
1403  * @param {URBGEN.Point[]} points - The points in which to look for the
1404  *     closest point.
1405  * @param {URBGEN.Point} target - The target point.
1406  * @return {URBGEN.Point} The point that is closest to the target.
1407  */
1408 URBGEN.Util.nearest = function (points, target) {
1409     var index = 0;
1410     var shortest = URBGEN.Util.getLineSegmentLength(points[0], target);
1411     for(var i = 1; i < points.length; i++) {
1412         var length = URBGEN.Util.getLineSegmentLength(points[i], target);
1413         if(length < shortest) {
1414             index = i;
1415             shortest = length;
1416         }
1417     }
1418     return points[index];
1419 };
